 	/**
+	 * Calculates delay in milliseconds until the next weekly respawn based on a configuration
+	 * string that may contain one or multiple entries, for example:
+	 * <ul>
+	 *   <li>{@code SUNDAY 18:00}</li>
+	 *   <li>{@code WEDNESDAY 11:00, SATURDAY 11:00}</li>
+	 * </ul>
+	 * 
+	 * @param respawnDate the configured respawn date string
+	 * @return delay in milliseconds until the next respawn, or {@code -1} if configuration is invalid
+	 */
+	public long getRespawnTime(String respawnDate)
+	{
+		if ((respawnDate == null) || respawnDate.isEmpty())
+		{
+			return -1;
+		}
+		
+		final Calendar now = Calendar.getInstance(TimeZone.getDefault(), Locale.getDefault());
+		long bestDelay = -1L;
+		
+		// Allow multiple entries separated by comma.
+		for (String token : respawnDate.split(","))
+		{
+			if (token == null)
+			{
+				continue;
+			}
+			
+			token = token.trim();
+			if (token.isEmpty())
+			{
+				continue;
+			}
+			
+			final String[] parts = token.split("\\s+");
+			if (parts.length != 2)
+			{
+				LOGGER.warning(getClass().getSimpleName() + ": Invalid grand boss respawn date token: " + token + " in " + respawnDate);
+				continue;
+			}
+			
+			final int dayOfWeek = parseDayOfWeek(parts[0]);
+			if (dayOfWeek == -1)
+			{
+				LOGGER.warning(getClass().getSimpleName() + ": Invalid grand boss respawn day token: " + token + " in " + respawnDate);
+				continue;
+			}
+			
+			final String[] timeParts = parts[1].split(":");
+			if (timeParts.length != 2)
+			{
+				LOGGER.warning(getClass().getSimpleName() + ": Invalid grand boss respawn time token: " + token + " in " + respawnDate);
+				continue;
+			}
+			
+			final int hour;
+			final int minute;
+			
+			try
+			{
+				hour = Integer.parseInt(timeParts[0]);
+				minute = Integer.parseInt(timeParts[1]);
+			}
+			catch (NumberFormatException e)
+			{
+				LOGGER.log(Level.WARNING, getClass().getSimpleName() + ": Invalid numbers in grand boss respawn time token: " + token + " in " + respawnDate, e);
+				continue;
+			}
+			
+			if ((hour < 0) || (hour > 23) || (minute < 0) || (minute > 59))
+			{
+				LOGGER.warning(getClass().getSimpleName() + ": Invalid hour/minute in grand boss respawn time token: " + token + " in " + respawnDate);
+				continue;
+			}
+			
+			final Calendar next = (Calendar) now.clone();
+			next.set(Calendar.DAY_OF_WEEK, dayOfWeek);
+			next.set(Calendar.HOUR_OF_DAY, hour);
+			next.set(Calendar.MINUTE, minute);
+			next.set(Calendar.SECOND, 0);
+			next.set(Calendar.MILLISECOND, 0);
+			
+			// If this week's slot already passed, move to next week.
+			if (!next.after(now))
+			{
+				next.add(Calendar.WEEK_OF_YEAR, 1);
+			}
+			
+			final long delay = next.getTimeInMillis() - now.getTimeInMillis();
+			if ((bestDelay == -1L) || (delay < bestDelay))
+			{
+				bestDelay = delay;
+			}
+		}
+		
+		if (bestDelay == -1L)
+		{
+			LOGGER.warning(getClass().getSimpleName() + ": No valid grand boss respawn entries in: " + respawnDate);
+		}
+		
+		return bestDelay;
+	}

+	/**
+	 * Parses a day-of-week string (e.g. {@code MONDAY}) into {@link Calendar} constants.
+	 * 
+	 * @param day the configured day string
+	 * @return the Calendar day-of-week value, or {@code -1} if invalid
+	 */
+	private int parseDayOfWeek(String day)
+	{
+		if (day == null)
+		{
+			return -1;
+		}
+		
+		final String value = day.trim().toUpperCase(Locale.ENGLISH);
+		
+		switch (value)
+		{
+			case "SUNDAY":
+			{
+				return Calendar.SUNDAY;
+			}
+			case "MONDAY":
+			{
+				return Calendar.MONDAY;
+			}
+			case "TUESDAY":
+			{
+				return Calendar.TUESDAY;
+			}
+			case "WEDNESDAY":
+			{
+				return Calendar.WEDNESDAY;
+			}
+			case "THURSDAY":
+			{
+				return Calendar.THURSDAY;
+			}
+			case "FRIDAY":
+			{
+				return Calendar.FRIDAY;
+			}
+			case "SATURDAY":
+			{
+				return Calendar.SATURDAY;
+			}
+		}
+		
+		return -1;
+	}
